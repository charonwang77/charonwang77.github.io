【C++】STL容器特性总结

Jun 1, 2021 · 3 min read

STL容器特性总结

1.Sequence Containers:

维持顺序的容器

vector:
    动态数组；
    O(1)随机读取，O(1)尾部增删；
    可作stack用。

 //创建vector容器对象，并且通过模板参数指定容器中存放的数据的类型
 vector<int> v1;
 //向容器中放数据
 v1.push_back(10);
 v1.push_back(20);
 v1.push_back(30);
 v1.push_back(40);
 
 vector<int> v2;
 v2 = v1;
 
 vector<int> v3;
 v3.assign(10,0);  // 10个0赋值给v3

 v1.begin();   // 返回迭代器，这个迭代器指向容器中第一个数据
 v1.end();     // 返回迭代器，这个迭代器指向容器元素的最后一个元素的下一个位置

 // 遍历方式：
 for (vector<int>::iterator it = v1.begin(); it != v1.end(); it++) 
 {
    cout << *it << endl;
 }
 cout << endl;

 // 功能:容量和大小
 v1.empty();    // 判断容器是否为空
 v1.capacity(); // 容器的容量
 v1.size();     // 返回容器中元素的个数
 v1.resize(10); // 重新指定容器的长度为num，若容器变长，则以默认值填充新位置。
 ​               // 如果容器变短，则末尾超出容器长度的元素被删除。
 v1.resize(10, 15); //重新指定容器的长度为num，若容器变长，则以elem值填充新位置。

 // 功能：插入和删除
 v1.push_back(50); //尾部插入元素ele
 v1.pop_back(); //删除最后一个元素
 v1.insert(v1.begin(), 60); //迭代器指向位置pos插入元素ele
 v1.insert(v1.begin(), 3, 70);//迭代器指向位置pos插入count个元素ele
 v1.erase(v1.begin()); //删除迭代器指向的元素
 v1.erase(v1.begin(), v1.end());//删除迭代器从start到end之间的元素
 v1.clear(); //删除容器中所有元素

 // 数据存取
 v1.at(3); //返回索引3所指的数据
 v1[3];    //返回索引3所指的数据
 v1.front(); //返回容器中第一个数据元素
 v1.back();  //返回容器中最后一个数据元素

 // 互换
 v1.swap(v2);
 // 预留空间
 v1.reseve(1000);

list :
    双向链表；
    可做stack和queue用。

 // 赋值
 list<int>L1;
 L1.push_back(10);
 L1.push_back(20);
 L1.push_back(30);
 L1.push_back(40);

 list<int>L2(L1.begin(),L1.end());

 list<int>L3(L2);

 list<int>L4(10, 1000);

 // 交换
 L1.swap(L4); //将lst与本身的元素互换。

 // 大小操作
 L1.size();  // 返回容器中元素的个数
 L1.empty(); // 判断容器是否为空
 L1.resize(10); //  重新指定容器的长度为10，若容器变长，则以默认值填充新位置。
                //  如果容器变短，则末尾超出容器长度的元素被删除。
 L1.resize(10, 100); //重新指定容器的长度为10，若容器变长，则以100值填充新位置。

 // 插入删除
 L1.push_back(100);  // 在容器尾部加入一个元素100
 L1.pop_back();      // 删除容器中最后一个元素
 L1.push_front(200); // 在容器开头插入一个元素200
 L1.pop_front();     // 从容器开头移除第一个元素
 
 list<int>::iterator it = L.begin();
 L.insert(++it, 1000);  // 在1位置插元素300的拷贝，返回新数据的位置。
 L1.insert(++it,4,300); // 在1位置插入4个300数据，无返回值。
 L1.insert(pos,beg,end); //在pos位置插入[beg,end)区间的数据，无返回值。
 L1.clear();         // 移除容器的所有数据
 
 L1.erase(beg,end);   // 删除[beg,end)区间的数据，返回下一个数据的位置。
 L1.erase(++it);      // 删除1位置的数据，返回下一个数据的位置。
 
 L1.remove(100);     // 删除容器中所有与elem值匹配的元素。

 // 反转和排序
 L1.reverse(); //反转链表
 L1.sort(); //链表排序

deque:
    双端队列；
    O(1)随机读取，O(1)头部尾部增删。

 // 赋值
 deque<int> d1; //无参构造函数
 d1.push_back(10);
 d1.push_back(20);
 d1.push_back(30);
 
 deque<int> d2(d1.begin(),d1.end());
 deque<int>d3(10,100);  //构造函数将n个elem拷贝给本身。
 deque<int>d4 = d3;

 // 大小操作
 d1.empty(); // 判断容器是否为空
 d1.size();  // 返回容器中元素的个数
 d1.resize(15); // 重新指定容器的长度为15,若容器变长，则以默认值填充新位置。
                    ​// 如果容器变短，则末尾超出容器长度的元素被删除。
 d1.resize(15, 100); // 重新指定容器的长度为15,若容器变长，则以100值填充新位置。
                          // 如果容器变短，则末尾超出容器长度的元素被删除。

 // 插入删除1
 d1.push_back(100); //在容器尾部添加一个数据
 d1.push_front(200); //在容器头部插入一个数据
 d1.pop_back(); //删除容器最后一个数据
 d1.pop_front(); //删除容器第一个数据

 // 插入删除2
 d1.insert(pos,elem); //在pos位置插入一个elem元素的拷贝，返回新数据的位置。
 d1.insert(pos,n,elem); //在pos位置插入n个elem数据，无返回值。
 d1.insert(pos,beg,end); //在pos位置插入[beg,end)区间的数据，无返回值。
 d1.clear(); //清空容器的所有数据
 d1.erase(beg,end); //删除[beg,end)区间的数据，返回下一个数据的位置。
 d1.erase(pos); //删除pos位置的数据，返回下一个数据的位置。

 // 数据存取
 d1.at(int idx); //返回索引idx所指的数据
 d1.[idx]; //返回索引idx所指的数据
 d1.front(); //返回容器中第一个数据元素
 d1.back(); //返回容器中最后一个数据元素

 // 排序
 sort(d1.begin(), d1.end());

2.Container Adaptors:

基于其他容器实现的数据结构

stack:
    栈，后入先出；
    常用于深度优先、字符串匹配、单调栈问题；
    默认基于deque。

 // 构造函数：
 stack<int> s1; //stack采用模板类实现， stack对象的默认构造形式
 s1.push(10);
 s1.push(20);
 s1.push(30); 
 stack<int> s2(s1); //拷贝构造函数  
 
 // 数据存取：  
 s1.push(40); // 向栈顶添加元素
 s1.pop();      // 从栈顶移除第一个元素
 s1.top();      // 返回栈顶元素
 
 //大小操作：
 s1.empty(); //判断堆栈是否为空
 s1.size(); //返回栈的大小

queue:
    队列，先入先出；
    常用于广度优先搜索；
    默认基于deque。

 // 构造函数：
 queue<int> q1; //queue采用模板类实现，queue对象的默认构造形式
 q1.push(10);
 q1.push(20);
 queue<int> q2(q1); //拷贝构造函数  
 
 // 数据存取
 q1.push(elem); //往队尾添加元素
 q1.pop(); //从队头移除第一个元素
 q1.back(); //返回最后一个元素
 q1.front(); //返回第一个元素
 
 // 大小操作：
 q1.empty(); //判断队列是否为空
 q1.size(); //返回队列的大小

priority_queue:
    最大值先出的数据结构；
    用O(nlogn)的时间排序数组，O(nlogn)的时间插入任意值，O(1)的时间获得最大值, O(logn)的时间删除最大值；
    常用于维护数据结构并快速获取最大最小；
    默认基于vector。

3.Associative Containers:

实现了排好序的数据结构

set:
    有序集合，元素不可重复；
    O(nlogn)排序数组，插入删除查找任意值，获得最大最小值。

 // 构造：
 set<int> st1; // 默认构造函数;
 st1.insert(10);
 st1.insert(20);
 st1.insert(30);
 st1.insert(40);
 
 set<int> st2;
 st2(st1);     // 拷贝构造函数  
 
 set<int>s3;
 s3 = s2;

 //大小和交换
 st1.size();  // 返回容器中元素的数目
 st1.empty(); // 判断容器是否为空
 st1.swap(st2); // 交换两个集合容器

 // 插入和删除
 st1.insert(elem); // 在容器中插入元素。
 st1.clear();      // 清除所有元素
 st1.erase(pos);   // 删除pos迭代器所指的元素，返回下一个元素的迭代器。
 st1.erase(beg, end); // 删除区间[beg,end)的所有元素 ，返回下一个元素的迭代器。
 st1.erase(elem);     // 删除容器中值为elem的元素。

 // 查找和统计
 st1.find(key);  // 查找key是否存在,若存在，返回该键的元素的迭代器；若不存在，返回set.end();
 st1.count(key); // 统计key的元素个数

multiset:
    支持重复元素的set。

map:
    有序映射或有序表；
    在map的基础上加上映射关系，可以对每一个元素key存一个值value。

 // 构造函数
 map<int,int> m1; // 默认构造
 m1.insert(pair<int, int>(1, 10));
 m1.insert(pair<int, int>(2, 20));
 m1.insert(pair<int, int>(3, 30)); 
 
 map<int, int> m2(m1); //拷贝构造  
 
 map<int, int>m3;
 m3 = m2; //赋值

// 大小和交换
m1.size();   // 返回容器中元素的数目
m1.empty();  // 判断容器是否为空
m1.swap(m2); // 交换两个集合容器

 // 插入和删除
 //插入
 map<int, int> m;
 //第一种插入方式
 m.insert(pair<int, int>(1, 10));
 //第二种插入方式
 m.insert(make_pair(2, 20));
 //第三种插入方式
 m.insert(map<int, int>::value_type(3, 30));
 //第四种插入方式
 m[4] = 40;
 
 //删除
 m.erase(m.begin()); //c删除迭代器所指的元素，返回下一个元素的迭代器。
 m.erase(3);  //c删除容器中值为3的元素。
 //清空
 m.erase(m.begin(),m.end());
 m.clear(); //c清除所有元素

 //查找
 map<int, int>::iterator pos = m.find(3); // 查找key是否存在,若存在，返回该键的元素的迭代器；若不存在，返回set.end();
 int num = m.count(3); // 统计key的元素个数

multimap:
    支持重复元素的map。

4.Unordered Associative Containers:

对每个Associative Containers实现了哈希版本

unordered_set:
    哈希集合；
    O(1)插入、查找、删除元素；
    常用于快速查询一个元素是否在这个容器内。

unordered_multiset:
    支持重复元素的unordered_set。

unordered_map:
    哈希映射或哈希表；
    在unordered_set基础上加上映射关系，可以对每一个元素key存一个值value。

unordered_multimap:
    支持重复元素的unordered_map。
